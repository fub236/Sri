1. Create a Dockerfile as below:

# cat Dockerfile
FROM ubuntu:latest

ENV trainer sri

RUN apt-get update && apt-get install -y curl
RUN apt-get update && apt-get install -y nginx

WORKDIR /var/www/html/
ADD index.html ./

EXPOSE 80

CMD ["nginx", "-g", "daemon off;"]

STOPSIGNAL SIGTERM
HEALTHCHECK CMD curl localhost:80
#
